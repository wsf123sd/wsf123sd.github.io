    int a[5]={1,3,5,4,2};

    //sort排序中要放两个指针，第一个是起始地址，第二个是结束地址，但是要给数组最后一个元素的下一个地址,sort默认为升序排列
sort(&a[0],&a[5]);     
for(int i=0; i<=4; i++)
    cout << a[i] << endl;

//reverse用法：sort+reverse可以降序排列
reverse(&a[0],&a[5]);
for(int i=0; i<=4; i++)
    cout << a[i] << endl;
return 0;
